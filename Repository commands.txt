Create new repository:

echo "# Helloworld-api" >> README.md
git init
git add README.md
git commit -m "first commit"
git branch -M main
git remote add origin https://github.com/ahmerarman/Helloworld-api.git
git push -u origin main

git remote add origin https://github.com/ahmerarman/Helloworld-api.git
git branch -M main
